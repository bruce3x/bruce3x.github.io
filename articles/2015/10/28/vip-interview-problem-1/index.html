唯品会笔试编程题

Date 2015-10-28 Category 面试

编写一个程序，创建三个线程，id分别为A, B, C, 各自打印十次自己的id，要求按照字母顺序输出，如“ABCABCABC...”

/**
 * Created by zero on 2015/10/17.
 * MutilThreadPrint
 */
public class Demo {
    final String[] ids = {"A", "B", "C"};
    int current = 0;

    public static void main(String[] args) {
        Demo demo = new Demo();

        System.out.println("Start...");
        for (int i = 0; i < demo.ids.length; i++) {
            new Thread(new Task(demo, i)).start();

        }
    }
}

public class Task implements Runnable {

    private final Demo demo;
    private int idIndex;

    public Task(Demo demo, int idIndex) {
        this.demo = demo;
        this.idIndex = idIndex;
    }

    @Override
    public void run() {

        int count = 0;
        while (count < 10) {

            synchronized (demo) {
                if (demo.current == idIndex) {
                    System.out.print(demo.ids[idIndex]);
                    count++;
                    demo.current = (++demo.current) % demo.ids.length;
                }
            }
        }
    }
}
