特征分析¶

性别 Sex¶

性别应该是一个比较重要的特征，一般人们更倾向于将生存的机会让给妇女儿童。

分别绘制男性、女性幸存者人数以及幸存率的柱状图。女性的幸存率高达 70%，远高于男性的 20%。可见性别是一个很重要的相关特征。

def draw_sex(ax, male, female):
    ax.set_xticks([0, 1])
    ax.set_xticklabels(['Male', 'Female'])
    ax.set_xlabel('Sex')
    
    # 男性幸存
    ax.bar(0, male[1], align='center', color='g',
           alpha=0.5, label='Survived')
    
    # 男性未幸存
    ax.bar(0, male[0], align='center', color='r',
           alpha=0.5, bottom=male[1], label='Not survived')
    
    # 女性幸存
    ax.bar(1, female[1], align='center', color='g',
           alpha=0.5)
    
    # 女性未幸存
    ax.bar(1, female[0], align='center', color='r',
           alpha=0.5, bottom=female[1])
    
    ax.grid()
    ax.legend(fontsize=8)
    
def sex_analysis(df):
    sex_of_survival = pd.crosstab(df['Survived'], df['Sex'])
    
    fig = plt.figure(figsize=(6, 6))
    
    # 男女性别幸存数量
    ax_of_count = fig.add_subplot(1, 2, 1)
    ax_of_count.set_title('Survival count of Sex')
    ax_of_count.set_ylabel('Survival count')
    draw_sex(
        ax_of_count, 
        sex_of_survival['male'], 
        sex_of_survival['female'], 
    )
    
    # 男女性别幸存率
    ax_of_rate = fig.add_subplot(1, 2, 2)
    ax_of_rate.set_title('Survival rate of Sex')
    ax_of_rate.set_ylabel('Survival rate')
    draw_sex(
        ax_of_rate,
        sex_of_survival['male'] / sex_of_survival['male'].sum(),
        sex_of_survival['female'] / sex_of_survival['female'].sum(),
    )
    
    plt.tight_layout() # 自动调整间距
    plt.show()

sex_analysis(train_data)

年龄 Age¶

和性别特征类似，人们通常会把逃生机会让给小孩，因此小孩的幸存率应该相对较高。

年龄分布在 0~80 的区间内，可以以 5 年为一个年龄区间，分别绘制幸存者的年龄分布直方图和频率分布直方图。可以看到，0~15岁的小孩的幸存率明显较高，高于了平均幸存率，反而成年人的幸存率偏低。其中[85, 85)区间幸存率达 100%，这应该算是一个异常值，这个区间内只有一位 80 岁老人并且幸存，可以不与考虑。

因此年龄是一个相关特征，可以给数据集添加一个新的特征 Child，以 15 岁作为阈值，将年龄特征二值化为 0 和 1。

def plot_age(ax, hist, bins_edge):  
    ax.bar(
        bins_edge[:-1], 
        hist, 
        width=5, 
        color='g', 
        alpha=0.5,
        label='Survival',
    )

    ax.set_xlabel('Age')
    
    ax.grid()

def age_analysis2(df):
    # 排除缺失值
    age = df[df['Age'].notnull()]['Age']
    age_of_survivied = age[df['Survived'] == 1]
    age_of_not_survivied = age[df['Survived'] == 0]
    
    # 未幸存者
    h_not_survived = hist, bins_edge = \
        np.histogram(age_of_not_survivied, bins=17, range=(0, 85))
    
    # 幸存者的分组
    h_survived = hist, bins_edge = \
        np.histogram(age_of_survivied, bins=17, range=(0, 85))
     
    
    fig = plt.figure(figsize=(6, 6))
    
    # 绘制各年龄段的幸存者数量
    ax_of_count = fig.add_subplot(1, 2, 1)
    ax_of_count.set_ylabel('Survival count')
    ax_of_count.set_title('Survival count of Age')
    plot_age(ax_of_count, hist, bins_edge)
    
    # 绘制各年龄段的幸存率
    ax_of_rate = fig.add_subplot(1, 2, 2)
    ax_of_rate.set_ylabel('Survival rate')
    ax_of_rate.set_title('Survival rate of Age')

    # 计算各分组的幸存率
    h1_survivied_rate = h_survived[0] * 1. / (h_survived[0] + h_not_survived[0])
    
    plot_age(ax_of_rate, h1_survivied_rate, bins_edge)
    
    # 平均幸存率
    average_rate = age_of_survivied.size * 1. / age.size
    plt.axhline(average_rate, color='r', label='Average survival rate')
    
    plt.tight_layout() # 自动调整间距
    plt.legend(fontsize=8, loc='upper left')
    plt.show()

age_analysis2(train_data)
